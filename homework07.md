query Orders($filter: OrdersFilterInput, $pagination: OrdersPaginationInput) {
  orders(filter: $filter, pagination: $pagination) {
    nodes {
      id
      orderStatus {
        name
      }
      createdAt
      orderItems {
        quantity
        product {
          id
          title
          price
        }
      }
    }
    pageInfo {
      offset
      limit
    }
    totalCount
  }
}
